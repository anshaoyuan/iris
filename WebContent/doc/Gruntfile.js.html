module.exports = function (grunt) {
    'use strict';

    grunt.initConfig({

        config: grunt.file.readJSON('sloth.conf.json'),

        jade: {
            compile: {
                expand: true,
                cwd: '<%= config.path.source %>',
                src: ['**/*.jade', '!**/_*.jade'],
                dest: '<%= config.path.public %>',
                ext: '.html',
                options: {
                    data: {
                        env: '<%= config.env %>',
                        date: '<%= config.date %>'
                    },
                    namespace: '<%= config.namespace %>'
                }
            }
        },

        sass: {
            options: {
                trace: true,
                lineNumbers: true,
                style: 'expanded',
                loadPath: '<%= config.vendor.bootstrap.load %>'
            },
            bootstrap: {
                src: '<%= config.vendor.bootstrap.sass %>',
                dest: '<%= config.path.public %>common/bootstrap.<%= config.date %>.css'
            },
            sloth: {
                expand: true,
                cwd: '<%= config.path.source %>/',
                src: ['**/*.scss', '!**/_*.scss'],
                dest: '<%= config.path.public %>/',
                ext: '.<%= config.date %>.css'
            }
        },

        copy: {
            sloth: {
                expand: true,
                cwd: '<%= config.path.source %>',
                src: '**/*.js',
                dest: '<%= config.path.public %>'
            }
        },

        concat: {
            options: {
                banner: '\'use strict\';\n',
                stripBanners: true
            },
            plugin: {
                options: { banner: '@charset "UTF-8";' },
                src: '<%= config.vendor.plugin %>',
                dest: '<%= config.path.public %>common/vendor.<%= config.date %>.css'
            },
            angular: {
                src: '<%= config.vendor.angular %>',
                dest: '<%= config.path.public %>common/angular.<%= config.date %>.js'
            },
            sloth: {
                src: '<%= config.path.sloth %>',
                dest: '<%= config.path.public %>sloth.<%= config.date %>.js',
                options: {
                    process: function (src, path) {
                        return '// Source: ' + path + '\n' +
                            src.replace(/(^|\n)[ \t]*('use strict'|"use strict");?\s*/g, '$1');
                    }
                }
            }
        },

        jsdoc: {
            doc: {
                src: ['<%= config.path.source %>'],
                dest: '<%= config.path.doc %>',
                options: { configure: "jsdoc.config.json" }
            }
        },

        watch: {
            options: {
                livereload: true
            },
            jade_template: {
                files: ['<%= config.path.source %>**/*.jade'],
                tasks: ['template']
            },
            style_bootstrap: {
                files: ['<%= config.vendor.bootstrap.base %>**/*.scss'],
                tasks: ['style:bootstrap']
            },
            style_plugin: {
                files: '<%= config.vendor.plugin %>',
                taskt: ['concat:plugin']
            },
            style_sloth: {
                files: ['<%= config.path.source %>**/*.scss'],
                tasks: ['style:<%= config.name %>']
            },
            script_angular: {
                files: '<%= config.vendor.angular %>',
                tasks: ['concat:angular']
            },
            script_sloth: {
                files: '<%= config.path.sloth %>',
                tasks: ['concat:sloth']
            }
        },

        connect: {
            server: {
                options: {
                    port: 1234,
                    open: {
                        appName: '<%= config.browser %>',
                        target: 'http://localhost:1234/public/'
                    },
                    livereload: true
                }
            },
            doc: {
                options: {
                    base: 'doc/',
                    port: 1235,
                    open: true,
                    keepalive: true
                }
            }
        }

    });

    grunt.loadNpmTasks('grunt-contrib-jade');
    grunt.loadNpmTasks('grunt-contrib-sass');
    grunt.loadNpmTasks('grunt-contrib-copy');
    grunt.loadNpmTasks('grunt-contrib-concat');
    grunt.loadNpmTasks('grunt-jsdoc');
    grunt.loadNpmTasks('grunt-contrib-watch');
    grunt.loadNpmTasks('grunt-contrib-connect');

    /**
     * 缺省任务，命令行下输入 grunt 默认执行此任务
     */
    grunt.registerTask('default', ['server', 'watch']);

    /**
     * 模版编译，通过传递 `env` 环境变量向模版传递不同的 `data`
     *
     * @global
     * @method grunt template
     * @param {string} env - 环境变量的名称
     * @default 'dev'
     * @example
     * // 开发环境，模版引未合并压缩的静态资源（默认）
     * grunt template
     *
     * // 发布环境，模版引经合并压缩的静态资源
     * grunt template:pub
     */
    grunt.registerTask('template', function (env) {
        if (env) {
            grunt.config.set('config.env', env);
            grunt.config.set('jade.compile.options.pretty', false);
        }
        grunt.config.set('jade.compile.options.pretty', true);
        grunt.task.run('jade:compile');
    });

    /**
     * 样式编译，env 决定是否压缩及文件后缀时间戳，task 指定子任务
     *
     * @global
     * @method grunt style
     * @param {string} task - 子任务的名称
     * @param {string} env - 环境变量的名称
     * @default 'dev'
     * @example
     * // 编译 Twitter Bootstrap，不压缩
     * grunt styel:bootstrap
     *
     * // 编译 Twitter Bootstrap，带压缩
     * grunt style:bootstrap:pub
     */
    grunt.registerTask('style', function (task, env) {
        var dest = grunt.template.process(
            '<%= config.path.public %>common/' + task + '.<%= config.date %>.css'
        );
        if (env) {
            grunt.config.set('sass.' + task + '.dest', dest);
            grunt.config.set('sass.options.lineNumbers', false);
            grunt.config.set('sass.options.style', 'compressed');
        }
        grunt.task.run('sass:' + task);
    });

    /**
     * 本地 web 服务器
     *
     * 例如：grunt server:firefox = 执行 connect:server，并使用 Firefox 浏览器打开
     */
    grunt.registerTask('server', function (browser) {
        if (browser) {
            grunt.config.set('config.browser', browser);
        } else {
            grunt.config.set('config.browser', 'Google Chrome');
        }
        grunt.task.run('connect:server');
    });

    /** 临时测试任务 */
    grunt.registerTask('test', function () {
        var path = grunt.config.get('config').vendor.angular;
        grunt.log.writeln(path);
    });
};
